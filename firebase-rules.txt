// Firebase Firestore Security Rules for Humanities Society
// Copy and paste these rules into Firebase Console > Firestore Database > Rules

rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    // Allow anyone to read newsletters
    match /newsletters/{document} {
      allow read: if true;
      // Only allow write operations (create, update, delete) - no authentication for now
      // In production, you should add authentication
      allow write: if true;
    }
    
    // Allow anyone to read events
    match /events/{document} {
      allow read: if true;
      // Only allow write operations (create, update, delete) - no authentication for now
      // In production, you should add authentication
      allow write: if true;
    }
    
    // Allow read/write for competition applications
    match /applications/{document} {
      allow read: if true;
      allow write: if true;
    }
    
    // Allow read/write for comments
    match /comments/{document} {
      allow read: if true;
      allow write: if true;
    }
  }
}

// NOTE: These rules are VERY PERMISSIVE for testing
// For production, you should:
// 1. Add Firebase Authentication
// 2. Restrict write access to authenticated users only
// 3. Add admin roles for delete operations
